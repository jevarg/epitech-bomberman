95  template <typename T>
96  GLM_FUNC_QUALIFIER tquat<T>::tquat
97  (
98  tvec3<T> const & eulerAngle
99  )
100  {
101  tvec3<T> c = glm::cos(eulerAngle * value_type(0.5));
102  tvec3<T> s = glm::sin(eulerAngle * value_type(0.5));
103 
104  this->w = c.x * c.y * c.z + s.x * s.y * s.z;
105  this->x = s.x * c.y * c.z - c.x * s.y * s.z;
106  this->y = c.x * s.y * c.z + s.x * c.y * s.z;
107  this->z = c.x * c.y * s.z - s.x * s.y * c.z;
108  }

447  template <typename T>
448  GLM_FUNC_QUALIFIER detail::tquat<T> mix
449  (
450  detail::tquat<T> const & x,
451  detail::tquat<T> const & y,
452  T const & a
453  )
454  {
455  T cosTheta = dot(x, y);
456 
457  // Perform a linear interpolation when cosTheta is close to 1 to avoid side effect of sin(angle) becoming a zero denominator
458  if(cosTheta > T(1) - epsilon<T>())
459  {
460  // Linear interpolation
461  return detail::tquat<T>(
462  mix(x.w, y.w, a),
463  mix(x.x, y.x, a),
464  mix(x.y, y.y, a),
465  mix(x.z, y.z, a));
466  }
467  else
468  {
469  // Essential Mathematics, page 467
470  T angle = acos(cosTheta);
471  return (sin((T(1) - a) * angle) * x + sin(a * angle) * y) / sin(angle);
472  }
473  }

492  (
493  detail::tquat<T> const & x,
494  detail::tquat<T> const & y,
495  T const & a
496  )
497  {

656  GLM_FUNC_QUALIFIER detail::tquat<T> quat_cast
657  (
658  detail::tmat3x3<T> const & m
659  )
660  {

662  typename detail::tquat<T>::value_type fourYSquaredMinus1 = m[1][1] - m[0][0] - m[2][2];
663  typename detail::tquat<T>::value_type fourZSquaredMinus1 = m[2][2] - m[0][0] - m[1][1];
664  typename detail::tquat<T>::value_type fourWSquaredMinus1 = m[0][0] + m[1][1] + m[2][2];
665 
666  int biggestIndex = 0;
667  typename detail::tquat<T>::value_type fourBiggestSquaredMinus1 = fourWSquaredMinus1;
668  if(fourXSquaredMinus1 > fourBiggestSquaredMinus1)
669  {
670  fourBiggestSquaredMinus1 = fourXSquaredMinus1;
671  biggestIndex = 1;
672  }
673  if(fourYSquaredMinus1 > fourBiggestSquaredMinus1)
674  {
675  fourBiggestSquaredMinus1 = fourYSquaredMinus1;
676  biggestIndex = 2;
677  }
678  if(fourZSquaredMinus1 > fourBiggestSquaredMinus1)
679  {
680  fourBiggestSquaredMinus1 = fourZSquaredMinus1;
681  biggestIndex = 3;
682  }
683 
684  typename detail::tquat<T>::value_type biggestVal = sqrt(fourBiggestSquaredMinus1 + T(1)) * T(0.5);
685  typename detail::tquat<T>::value_type mult = T(0.25) / biggestVal;
686 
687  detail::tquat<T> Result;
688  switch(biggestIndex)
689  {
690  case 0:
691  Result.w = biggestVal;
692  Result.x = (m[1][2] - m[2][1]) * mult;
693  Result.y = (m[2][0] - m[0][2]) * mult;
694  Result.z = (m[0][1] - m[1][0]) * mult;
695  break;
696  case 1:
697  Result.w = (m[1][2] - m[2][1]) * mult;
698  Result.x = biggestVal;
699  Result.y = (m[0][1] + m[1][0]) * mult;
700  Result.z = (m[2][0] + m[0][2]) * mult;
701  break;
702  case 2:
703  Result.w = (m[2][0] - m[0][2]) * mult;
704  Result.x = (m[0][1] + m[1][0]) * mult;
705  Result.y = biggestVal;
706  Result.z = (m[1][2] + m[2][1]) * mult;
707  break;
708  case 3:
709  Result.w = (m[0][1] - m[1][0]) * mult;
710  Result.x = (m[2][0] + m[0][2]) * mult;
711  Result.y = (m[1][2] + m[2][1]) * mult;
712  Result.z = biggestVal;
713  break;
714 
715  default: // Silence a -Wswitch-default warning in GCC. Should never actually get here. Assert is just for sanity.
716  assert(false);
717  break;
718  }
719  return Result;
