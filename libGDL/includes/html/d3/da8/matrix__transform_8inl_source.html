44  GLM_FUNC_QUALIFIER detail::tmat4x4<T> rotate
45  (
46  detail::tmat4x4<T> const & m,
47  T const & angle,
48  detail::tvec3<T> const & v
49  )
50  {
51 #ifdef GLM_FORCE_RADIANS
52  T a = angle;
53 #else
54  T a = radians(angle);
55 #endif
56  T c = cos(a);
57  T s = sin(a);
58 
59  detail::tvec3<T> axis = normalize(v);
60 
61  detail::tvec3<T> temp = (T(1) - c) * axis;
62 
63  detail::tmat4x4<T> Rotate(detail::tmat4x4<T>::null);
64  Rotate[0][0] = c + temp[0] * axis[0];
65  Rotate[0][1] = 0 + temp[0] * axis[1] + s * axis[2];
66  Rotate[0][2] = 0 + temp[0] * axis[2] - s * axis[1];
67 
68  Rotate[1][0] = 0 + temp[1] * axis[0] - s * axis[2];
69  Rotate[1][1] = c + temp[1] * axis[1];
70  Rotate[1][2] = 0 + temp[1] * axis[2] + s * axis[0];
71 
72  Rotate[2][0] = 0 + temp[2] * axis[0] + s * axis[1];
73  Rotate[2][1] = 0 + temp[2] * axis[1] - s * axis[0];
74  Rotate[2][2] = c + temp[2] * axis[2];
75 
76  detail::tmat4x4<T> Result(detail::tmat4x4<T>::null);
77  Result[0] = m[0] * Rotate[0][0] + m[1] * Rotate[0][1] + m[2] * Rotate[0][2];
78  Result[1] = m[0] * Rotate[1][0] + m[1] * Rotate[1][1] + m[2] * Rotate[1][2];
79  Result[2] = m[0] * Rotate[2][0] + m[1] * Rotate[2][1] + m[2] * Rotate[2][2];
80  Result[3] = m[3];
81  return Result;
82  }

119  template <typename T>
120  GLM_FUNC_QUALIFIER detail::tmat4x4<T> rotate_slow
121  (
122  detail::tmat4x4<T> const & m,
123  T const & angle,
124  detail::tvec3<T> const & v
125  )
126  {
127 #ifdef GLM_FORCE_RADIANS
128  T const a = angle;
129 #else
130  T const a = radians(angle);
131 #endif
132  T c = cos(a);
133  T s = sin(a);
134  detail::tmat4x4<T> Result;
135 
136  detail::tvec3<T> axis = normalize(v);
137 
138  Result[0][0] = c + (1 - c) * axis.x * axis.x;
139  Result[0][1] = (1 - c) * axis.x * axis.y + s * axis.z;
140  Result[0][2] = (1 - c) * axis.x * axis.z - s * axis.y;
141  Result[0][3] = 0;
142 
143  Result[1][0] = (1 - c) * axis.y * axis.x - s * axis.z;
144  Result[1][1] = c + (1 - c) * axis.y * axis.y;
145  Result[1][2] = (1 - c) * axis.y * axis.z + s * axis.x;
146  Result[1][3] = 0;
147 
148  Result[2][0] = (1 - c) * axis.z * axis.x + s * axis.y;
149  Result[2][1] = (1 - c) * axis.z * axis.y - s * axis.x;
150  Result[2][2] = c + (1 - c) * axis.z * axis.z;
151  Result[2][3] = 0;
152 
153  Result[3] = detail::tvec4<T>(0, 0, 0, 1);
154  return m * Result;
155  }

172  GLM_FUNC_QUALIFIER detail::tmat4x4<valType> ortho
173  (
174  valType const & left,
175  valType const & right,
176  valType const & bottom,

179  valType const & zFar
180  )
181  {

230  template <typename valType>
231  GLM_FUNC_QUALIFIER detail::tmat4x4<valType> perspective
232  (
233  valType const & fovy,
234  valType const & aspect,
235  valType const & zNear,
236  valType const & zFar
237  )
238  {
239  assert(aspect != valType(0));
240  assert(zFar != zNear);
241 
242 #ifdef GLM_FORCE_RADIANS
243  valType const rad = fovy;
244 #else
245  valType const rad = glm::radians(fovy);
246 #endif
247 
248  valType tanHalfFovy = tan(rad / valType(2));
249  detail::tmat4x4<valType> Result(valType(0));
250  Result[0][0] = valType(1) / (aspect * tanHalfFovy);
251  Result[1][1] = valType(1) / (tanHalfFovy);
252  Result[2][2] = - (zFar + zNear) / (zFar - zNear);
253  Result[2][3] = - valType(1);
254  Result[3][2] = - (valType(2) * zFar * zNear) / (zFar - zNear);
255  return Result;
256  }

287  (
288  T fovy,

339  template <typename T, typename U>
340  GLM_FUNC_QUALIFIER detail::tvec3<T> project
341  (
342  detail::tvec3<T> const & obj,
343  detail::tmat4x4<T> const & model,
344  detail::tmat4x4<T> const & proj,
345  detail::tvec4<U> const & viewport
346  )
347  {
348  detail::tvec4<T> tmp = detail::tvec4<T>(obj, T(1));
349  tmp = model * tmp;
350  tmp = proj * tmp;
351 
352  tmp /= tmp.w;
353  tmp = tmp * T(0.5) + T(0.5);
354  tmp[0] = tmp[0] * T(viewport[2]) + T(viewport[0]);
355  tmp[1] = tmp[1] * T(viewport[3]) + T(viewport[1]);
356 
357  return detail::tvec3<T>(tmp);
358  }

360  template <typename T, typename U>
361  GLM_FUNC_QUALIFIER detail::tvec3<T> unProject
362  (
363  detail::tvec3<T> const & win,
364  detail::tmat4x4<T> const & model,
365  detail::tmat4x4<T> const & proj,
366  detail::tvec4<U> const & viewport
367  )
368  {
369  detail::tmat4x4<T> inverse = glm::inverse(proj * model);
370 
371  detail::tvec4<T> tmp = detail::tvec4<T>(win, T(1));
372  tmp.x = (tmp.x - T(viewport[0])) / T(viewport[2]);
373  tmp.y = (tmp.y - T(viewport[1])) / T(viewport[3]);
374  tmp = tmp * T(2) - T(1);
375 
376  detail::tvec4<T> obj = inverse * tmp;
377  obj /= obj.w;
378 
379  return detail::tvec3<T>(obj);
380  }

382  template <typename T, typename U>
383  detail::tmat4x4<T> pickMatrix
384  (
385  detail::tvec2<T> const & center,
386  detail::tvec2<T> const & delta,
387  detail::tvec4<U> const & viewport
388  )
389  {
390  assert(delta.x > T(0) && delta.y > T(0));
391  detail::tmat4x4<T> Result(1.0f);
392 
393  if(!(delta.x > T(0) && delta.y > T(0)))
394  return Result; // Error
395 
396  detail::tvec3<T> Temp(
397  (T(viewport[2]) - T(2) * (center.x - T(viewport[0]))) / delta.x,
398  (T(viewport[3]) - T(2) * (center.y - T(viewport[1]))) / delta.y,
399  T(0));
400 
401  // Translate and scale the picked region to the entire window
402  Result = translate(Result, Temp);
403  return scale(Result, detail::tvec3<T>(T(viewport[2]) / delta.x, T(viewport[3]) / delta.y, T(1)));

406  template <typename T>
407  GLM_FUNC_QUALIFIER detail::tmat4x4<T> lookAt
408  (
409  detail::tvec3<T> const & eye,
410  detail::tvec3<T> const & center,
411  detail::tvec3<T> const & up
412  )
413  {
414  detail::tvec3<T> f = normalize(center - eye);
415  detail::tvec3<T> u = normalize(up);
416  detail::tvec3<T> s = normalize(cross(f, u));
417  u = cross(s, f);
418 
419  detail::tmat4x4<T> Result(1);
420  Result[0][0] = s.x;
421  Result[1][0] = s.y;
422  Result[2][0] = s.z;
423  Result[0][1] = u.x;
424  Result[1][1] = u.y;
425  Result[2][1] = u.z;
426  Result[0][2] =-f.x;
427  Result[1][2] =-f.y;
428  Result[2][2] =-f.z;
429  Result[3][0] =-dot(s, eye);
430  Result[3][1] =-dot(u, eye);
431  Result[3][2] = dot(f, eye);
432  return Result;
433  }
