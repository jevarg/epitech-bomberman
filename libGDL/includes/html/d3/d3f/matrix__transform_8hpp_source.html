38 /// conventions. For example, the lookAt function generates a transform from world
39 /// space into the specific eye space that the projective matrix functions
40 /// (perspective, ortho, etc) are designed to expect. The OpenGL compatibility
41 /// specifications defines the particular layout of this eye space.

61  /// Builds a translation 4 * 4 matrix created from a vector of 3 components.
62  ///
63  /// @param m Input matrix multiplied by this translation matrix.
64  /// @param v Coordinates of a translation vector.
65  /// @tparam T Value type used to build the matrix. Currently supported: half (not recommanded), float or double.
66  /// @code
67  /// #include <glm/glm.hpp>
68  /// #include <glm/gtc/matrix_transform.hpp>
69  /// ...
70  /// glm::mat4 m = glm::translate(glm::mat4(1.0f), glm::vec3(1.0f));
71  /// // m[0][0] == 1.0f, m[0][1] == 0.0f, m[0][2] == 0.0f, m[0][3] == 0.0f
72  /// // m[1][0] == 0.0f, m[1][1] == 1.0f, m[1][2] == 0.0f, m[1][3] == 0.0f
73  /// // m[2][0] == 0.0f, m[2][1] == 0.0f, m[2][2] == 1.0f, m[2][3] == 0.0f
74  /// // m[3][0] == 1.0f, m[3][1] == 1.0f, m[3][2] == 1.0f, m[3][3] == 1.0f
75  /// @endcode
76  /// @see gtc_matrix_transform
