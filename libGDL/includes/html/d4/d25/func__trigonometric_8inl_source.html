201  (

203  )
204  {
205  GLM_STATIC_ASSERT(detail::type<genType>::is_float, "'asinh' only accept floating-point input");
206 
207  return (x < genType(0) ? genType(-1) : (x > genType(0) ? genType(1) : genType(0))) * log(abs(x) + sqrt(genType(1) + x * x));
